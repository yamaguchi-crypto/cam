// ============================================================ // 楽天レビュー要約機能（src/index.js に追加するモジュール） // ============================================================ // // 【wrangler.jsonc に追加が必要な設定】 // "ai": { "binding": "AI" } // （Workers AI を env.AI で使えるようにするバインディング） // // 【npm 依存（EUC-JP保険用 / 純JS・Workers動作可）】 // npm i encoding-japanese // // 【ルーティングへの組み込み（既存の fetch ハンドラ内に1行追加）】 // if (url.pathname === "/api/summarize-review" && request.method === "GET") { // return handleSummarizeReview(request, env); // } // ============================================================ import Encoding from "encoding-japanese"; // --- 設定（環境に合わせて調整） --------------------------------- const AI_MODEL = "@cf/meta/llama-3.1-8b-instruct"; // ↑ モデルIDは https://developers.cloudflare.com/workers-ai/models/ で最新を確認。 // レビュー件数を多く入れたい場合は 32k コンテキストの Mistral 系に変更すると安定: // 例: "@cf/mistralai/mistral-7b-instruct-v0.2" const MAX_PAGES = 2; // 取得する楽天レビューのページ数（ニューロン節約のため少なめ推奨） const REVIEW_CHAR_LIMIT = 3000; // AIに渡すレビュー本文の最大文字数（8Bは4096トークン上限なので控えめ） const MAX_REVIEWS = 40; // 取り込む最大レビュー件数 // ============================================================ // ① charset自動判定つき fetch（UTF-8/EUC-JP/Shift_JIS対応） // ============================================================ async function fetchHtmlSmart(targetUrl) { const res = await fetch(targetUrl, { headers: { "User-Agent": "Mozilla/5.0 (compatible; ReviewSummaryBot/1.0)", "Accept-Language": "ja,en;q=0.8", "Accept": "text/html", }, }); if (!res.ok) throw new Error("ページ取得失敗 HTTP " + res.status); const buf = new Uint8Array(await res.arrayBuffer()); // charset判定: Content-Type ヘッダ → HTML内のmetaタグ の順 const ctype = res.headers.get("content-type") || ""; let charset = (ctype.match(/charset=([\w-]+)/i)?.[1] || "").toLowerCase(); if (!charset) { const head = new TextDecoder("utf-8", { fatal: false }).decode(buf.slice(0, 2048)); charset = (head.match(/charset=["']?([\w-]+)/i)?.[1] || "utf-8").toLowerCase(); } // UTF-8 はそのまま（Workers標準でOK） if (charset === "utf-8" || charset === "utf8" || charset === "") { return new TextDecoder("utf-8").decode(buf); } // EUC-JP / Shift_JIS は encoding-japanese で変換（Workers標準TextDecoderは非対応のため） const from = /euc/.test(charset) ? "EUCJP" : /(shift|sjis|s_jis|x-sjis|ms_kanji|windows-31j)/.test(charset) ? "SJIS" : "AUTO"; return Encoding.convert(buf, { to: "UNICODE", from, type: "string" }); } // ============================================================ // ② レビュー本文の抽出（ハッシュ化クラス名に依存しない方式） // 楽天レビューの安定UI文字列「不適切レビュー報告」「ショップからのコメント」で区切る // ============================================================ function decodeEntities(s) { return s .replace(/ /g, " ") .replace(/&/g, "&") .replace(/</g, "<") .replace(/>/g, ">") .replace(/"/g, '"') .replace(/'/g, "'") .replace(/&#x([0-9a-f]+);/gi, (_, h) => String.fromCodePoint(parseInt(h, 16))) .replace(/&#(\d+);/g, (_, d) => String.fromCodePoint(parseInt(d, 10))); } function extractReviews(html) { // script/style/noscript を除去 let s = html .replace(//gi, " ") .replace(//gi, " ") .replace(//gi, " "); // タグをスペースに、HTMLエンティティを復元 s = decodeEntities(s.replace(/<[^>]+>/g, " ")); s = s.replace(/[ \t\f\v\u00a0]+/g, " ").replace(/ *\n */g, "\n").trim(); // レビュー一覧の領域に絞る（ヘッダ・公式AI要約・フッタのノイズを落とす） const startIdx = s.indexOf("レビュー一覧"); if (startIdx >= 0) s = s.slice(startIdx + "レビュー一覧".length); for (const marker of ["お客様のご意見をお聞かせください", "楽天市場の関連ジャンル", "レビューをご利用になる際"]) { const i = s.indexOf(marker); if (i >= 0) { s = s.slice(0, i); break; } } // 「不適切レビュー報告」が各レビューの末尾に1回出るのでそこで分割 const blocks = s.split("不適切レビュー報告"); const reviews = []; for (let b of blocks) { b = b.trim(); if (b.length < 15) continue; // ショップ返信は要約対象外 → 「ショップからのコメント」以降を切り落とす const shopIdx = b.indexOf("ショップからのコメント"); if (shopIdx >= 0) b = b.slice(0, shopIdx).trim(); // UI文言のノイズを除去 b = b .replace(/さらに表示/g, " ") .replace(/参考になった/g, " ") .replace(/\d+人\s*が参考になったと回答/g, " ") .replace(/\s+/g, " ") .trim(); if (b.length >= 15) reviews.push(b); if (reviews.length >= MAX_REVIEWS) break; } return reviews; } // ============================================================ // ③ ページ番号を差し替えて複数ページ収集 // URL例: https://review.rakuten.co.jp/item/1/270424_10048681/1.1/ // 末尾の「1.1」= ページ番号.ソート種別 // ============================================================ function buildPageUrl(baseUrl, page) { return baseUrl.replace(/\/(\d+)\.(\d+)\/?$/, `/${page}.$2/`); } async function collectReviews(baseUrl) { const all = []; for (let p = 1; p <= MAX_PAGES; p++) { const pageUrl = buildPageUrl(baseUrl, p); try { const html = await fetchHtmlSmart(pageUrl); const rv = extractReviews(html); if (rv.length === 0) break; // これ以上ページがない all.push(...rv); if (all.length >= MAX_REVIEWS) break; } catch (e) { if (p === 1) throw e; // 1ページ目の失敗は致命的 break; // 2ページ目以降の失敗は無視して続行 } } return all.slice(0, MAX_REVIEWS); } // ============================================================ // ④ Workers AI で要約（メリット/デメリット/改善点をJSONで） // ============================================================ function safeJsonParse(text) { if (!text) return null; // コードブロックの除去＋最初の { 〜 最後の } を抽出 let t = String(text).replace(/```json/gi, "").replace(/```/g, "").trim(); const start = t.indexOf("{"); const end = t.lastIndexOf("}"); if (start >= 0 && end > start) t = t.slice(start, end + 1); try { return JSON.parse(t); } catch { return null; } } async function summarizeReviews(env, reviewText) { const system = `あなたは家具ECの商品分析アシスタントです。与えられた楽天市場の商品レビュー本文を分析し、必ず下記スキーマのJSONのみを日本語で出力してください。前置き・後置き・コードブロック・説明文は一切出力しないこと。 スキーマ: { "merits": ["顧客が満足している点・良かった点を5〜8個"], "demerits": ["不満点・悪かった点を3〜6個。なければ空配列[]"], "improvements": ["商品開発・改良に活かせる具体的な改善提案を3〜6個。『何が問題か→どう直すか』をセットで"] } ルール: - レビュー本文に書かれている事実のみを根拠にする。推測で水増ししない。 - 同じ趣旨の声は1項目にまとめ、言及の多い順に並べる。 - 各項目は簡潔な日本語1文にする。`; const user = `次の商品レビューを分析してJSONで出力してください。\n\n${reviewText}`; const resp = await env.AI.run(AI_MODEL, { messages: [ { role: "system", content: system }, { role: "user", content: user }, ], max_tokens: 1024, temperature: 0.2, }); const parsed = safeJsonParse(resp?.response); if (!parsed) { // モデルがJSONを崩した場合のフォールバック（生テキストを返す） return { merits: [], demerits: [], improvements: [], _raw: resp?.response ?? "" }; } return { merits: Array.isArray(parsed.merits) ? parsed.merits : [], demerits: Array.isArray(parsed.demerits) ? parsed.demerits : [], improvements: Array.isArray(parsed.improvements) ? parsed.improvements : [], }; } // ============================================================ // ⑤ ルートハンドラ（既存 fetch から呼ぶ） // ============================================================ export async function handleSummarizeReview(request, env) { const json = (obj, status = 200) => new Response(JSON.stringify(obj), { status, headers: { "Content-Type": "application/json; charset=utf-8", "Cache-Control": "no-store" }, }); try { const reqUrl = new URL(request.url); const target = reqUrl.searchParams.get("url"); if (!target) return json({ ok: false, error: "url パラメータが必要です" }, 400); // --- SSRF対策: 楽天レビューのドメインのみ許可 --- let parsed; try { parsed = new URL(target); } catch { return json({ ok: false, error: "URLが不正です" }, 400); } if (parsed.hostname !== "review.rakuten.co.jp") { return json({ ok: false, error: "review.rakuten.co.jp のURLのみ対応しています" }, 400); } const reviews = await collectReviews(target); if (reviews.length === 0) { return json({ ok: false, error: "レビュー本文を抽出できませんでした" }, 422); } // トークン予算内に収める（先頭=新しい順から詰める） let text = ""; let used = 0; for (const r of reviews) { if (used + r.length > REVIEW_CHAR_LIMIT) break; text += "・" + r + "\n"; used += r.length; } const summary = await summarizeReviews(env, text); return json({ ok: true, source: target, reviewCountUsed: text.split("\n").filter(Boolean).length, summary, // { merits, demerits, improvements } }); } catch (e) { return json({ ok: false, error: e.message }, 500); } }
